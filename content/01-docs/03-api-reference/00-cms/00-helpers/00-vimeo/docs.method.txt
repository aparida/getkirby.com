Title: vimeo
----
Syntax: vimeo($url, $width = false, $height = false, $class = false)
----
Description: Embeds a Vimeo video by url
----
Text: 
----
Parameters: - name: $url
  type: string
  text: The Vimeo url, ie. http://vimeo.com/52345557

- name: $width
  type: int
  text: The width of the embedded video

- name: $height
  type: int
  text: The height of the embedded video

- name: $class
  type: string
  text: an additional class selector which should be added to the iframe
----
Returntype: string
----
Return: The generated html
