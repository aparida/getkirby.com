Title: content::load

----
Text: 
----
Parameters: - name: $file
  type: string
  text: The path to the file

- name: $data
  type: array
  text: Additional variables which should be available for the loaded content

- name: $return
  type: boolean
  text: True: return the content of the file, false: echo the content
