----
Text: 
----
Parameters: - name: $rules
  type: array
  text: A set of rules to validate the model by

- name: $messages
  type: array
  text: An optional set of error messages for each used validation method

- name: $attributes
  type: array
  text: An optional set of attribute names to be used in error messages
